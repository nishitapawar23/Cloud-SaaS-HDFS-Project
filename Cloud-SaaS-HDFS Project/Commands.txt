# Start Hadoop services
start-dfs.sh

# Check HDFS directories
hdfs dfs -ls /

# Create cloud directory
hdfs dfs -mkdir /cloud

# Upload file to HDFS (simulate cloud upload)
hdfs dfs -put demo.txt /cloud

# List files inside cloud directory
hdfs dfs -ls /cloud

# View file content
hdfs dfs -cat /cloud/demo.txt

# Download file from HDFS (simulate cloud download)
hdfs dfs -get /cloud/demo.txt

# Delete file from HDFS (optional)
hdfs dfs -rm /cloud/demo.txt
